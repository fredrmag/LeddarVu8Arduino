121 // REGMAP_TRN_CGF - R/W - 0xFFFB00
122 #define LEDDARVU8_SECURE_TRANSACTION_ENABLED_FLAGS 0x00 | REGMAP_TRN_CGF // Secure-transaction enabled flags:
123 //1 = Enables the CRC calculation and validation on any transaction. This flag is enabled by default.
124 //0 = No CRC validation. The CRC field is still required in SPI protocol but can be set to any value.
125 #define LEDDARVU8_TRANSACTION_MODE 0x01 | REGMAP_TRN_CGF // Transaction modes:
126 // 0 = Free run. The READY pin is asserted on each ready detection frame. The host must be able to read data on time.
127 // 1 = Blocking read. On the READY pin assertion, host must read all data from traces or detections bank (data transaction control source configuration) to continue acquisition.
128 // 2 = Partial blocking read. On the READY pin assertion, host can read all data from traces of the detection bank and the acquisition is still running. Possible loss of detection frames if the host reading data is very long.
129 #define LEDDARVU8_CRC_OF_THE_LAST_TRANSACTION 0x02 | REGMAP_TRN_CGF // CRC of the last transaction
130 #define LEDDARVU8_INFORMATION_LAST_TRANSACTION 0x04 | REGMAP_TRN_CGF // Bit-field information of last transactions:
131 //All bits to 0: No transaction error
132 // Bit-0: Access right violation
133 // Bit-1: Invalid address
134 // Bit-2: Command not found
135 // Bit-3: Write disabled
136 // Bit-4: CRC failed
137 // Bit-5: Command execution error
138 // Bit-6: Invalid packet
139 #define LEDDARVU8_DATA_TRANSACTION_CONTROL_SOURCE 0x06 | REGMAP_TRN_CGF // Data transaction control source:
140 // 0 = On trace
141 // 1 = On detections
